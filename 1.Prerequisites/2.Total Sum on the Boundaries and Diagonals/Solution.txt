#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int** arr=new int*[n];
    for(int i=0;i<n;i++)
    {
        arr[i]=new int[n];
        for(int j=0;j<n;j++)
		cin>>arr[i][j];
    }
    int sum=0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if((i==0)||(j==0)||(i==n-1)||(j==n-1)||(i==j)||(j==n-i-1))
                sum=sum+arr[i][j];
        }
    }
    cout<<sum<<endl;
	return 0;
}
